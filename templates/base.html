from datetime import datetime from zoneinfo import ZoneInfo CITIES = [ { "name": "Mexico City", "timezone": "America/Mexico_City" }, { "name": "New York", "timezone": "America/New_York" }, { "name": "London", "timezone": "Europe/London" }, { "name": "Tokyo", "timezone": "Asia/Tokyo" } ] def get_city_times(): results = [] for city in CITIES: now = datetime.now(ZoneInfo(city["timezone"])) timeline = [] for hour in range(24): timeline.append({ "hour": hour, "label": f"{hour:02d}" }) results.append({ "city": city["name"], "time": now.strftime("%H:%M"), "date": now.strftime("%a %d %b"), "timeline": timeline }) return results
